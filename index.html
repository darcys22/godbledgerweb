--Trial Balance directly from MySQL
SELECT
  split_accounts.account_id,
  SUM(splits.amount)
FROM splits
JOIN split_accounts
  ON splits.split_id = split_accounts.split_id
WHERE splits.split_date <= CURDATE()
GROUP BY split_accounts.account_id
